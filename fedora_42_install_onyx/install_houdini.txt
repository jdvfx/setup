# install Nvidia drivers and Cuda+OpenCL
sudo dnf update
sudo reboot

sudo dnf install akmod-nvidia

# check if nvidia drivers are installed
lsmod | grep -i nvi

# check if nouveau drivers are gone
lsmod | grep -i nou

# check nvidia driver version:
nvidia-smi

# enable login screen with Xorg (if needed)
vi /etc/gdm/custom.conf

# search for available versions of the drivers
dnf list --showduplicates *nvidia*
